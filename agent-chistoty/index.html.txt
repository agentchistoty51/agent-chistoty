        function addToCart(serviceId) {
            const existing = cart.find(i => i.id === serviceId);
            if (existing) existing.quantity++;

        function updateCartQuantity(serviceId, delta) {
            const item = cart.find(i => i.id === serviceId);
            if (!item) return;
            item.quantity += delta;
            if (item.quantity <= 0) removeFromCart(serviceId);
            else { saveCart(); renderCart(); }
        }

        function renderCalendar() {
            const container = document.getElementById('calendarWidget');
            if (!container) return;
            const monthNames = ['Январь','Февраль','Март','Апрель','Май','Июнь','Июль','Август','Сентябрь','Октябрь','Ноябрь','Декабрь'];
            document.getElementById('clientMonthLabel').textContent = `${monthNames[clientCurrentMonth]} ${clientCurrentYear}`;
            const daysInMonth = new Date(clientCurrentYear, clientCurrentMonth + 1, 0).getDate();
            const firstDayOfWeek = new Date(clientCurrentYear, clientCurrentMonth, 1).getDay();
            const today = new Date(); today.setHours(0,0,0,0);
            let html = `<div class="grid grid-cols-7 gap-1 text-xs text-gray-500 mb-1">` + ['Вс','Пн','Вт','Ср','Чт','Пт','Сб'].map(d => `<div class="text-center">${d}</div>`).join('') + `</div><div class="grid grid-cols-7 gap-1">`;
            for (let i = 0; i < firstDayOfWeek; i++) html += `<div class="calendar-day empty"></div>`;
            for (let day = 1; day <= daysInMonth; day++) {
                const dateStr = `${clientCurrentYear}-${String(clientCurrentMonth+1).padStart(2,'0')}-${String(day).padStart(2,'0')}`;
                const currentDate = new Date(clientCurrentYear, clientCurrentMonth, day);
                const isPast = currentDate < today;
                const bookedTimes = getBookedTimesForDate(dateStr);
                const fullyBooked = bookedTimes.length === 10;
                const hasBooked = bookedTimes.length > 0;
                const isSelected = selectedDate === dateStr;
                let cls = 'calendar-day';
                if (isPast) cls += ' past';
                else if (fullyBooked) cls += ' booked';
                else if (hasBooked && !fullyBooked) cls += ' partial';
                if (isSelected && !isPast && !fullyBooked) cls += ' selected';
                const clickHandler = (isPast || fullyBooked) ? '' : `onclick="selectClientDate('${dateStr}')"`;
                html += `<div class="${cls}" data-date="${dateStr}" ${clickHandler}>${day}</div>`;
            }
            html += `</div>`;
            container.innerHTML = html;
            renderClientTimeSlots();
        }
        function selectClientDate(dateStr) {

            renderCalendar(); renderClientTimeSlots();

            if (!input.files || !input.files[0]) return;

            const file = input.files[0];

                    document.getElementById('aiUpload').classList.add('hidden');

                    const predictions = await cocoModel.detect(img);

        function finishAIAnalysis(predictions, img) {
            updateProgress(100, 'Готово!');
            document.getElementById('aiProgress').classList.add('hidden');
            const detectedObjects = predictions.filter(p => p.score > 0.5).map(p => p.class);
            const uniqueObjects = [...new Set(detectedObjects)];
            const canvas = document.getElementById('analysisCanvas');
            const ctx = canvas.getContext('2d');
            canvas.width = 200; canvas.height = 150;
            ctx.drawImage(img, 0, 0, 200, 150);
            const imageData = ctx.getImageData(0, 0, 200, 150);
            const data = imageData.data;
            let highSatPixels = 0;
            for (let i = 0; i < data.length; i += 4) {
                const r = data[i], g = data[i+1], b = data[i+2];
                const max = Math.max(r,g,b), min = Math.min(r,g,b);
                const saturation = max === 0 ? 0 : (max - min) / max;
                if (saturation > 0.7) highSatPixels++;
            }
            const stainDetected = highSatPixels > 0.03 * (data.length / 4);
            let recommended = [];
            uniqueObjects.forEach(obj => {
                services.forEach(s => {
                    if (s.aiObjects.includes(obj) && !recommended.find(r => r.id === s.id)) {
                        recommended.push(s);
                    }
                });
            });
            if (recommended.length === 0) recommended.push(getServiceById('gen_apt'));
            const resultDiv = document.getElementById('aiResult');
            resultDiv.innerHTML = `
                <div class="bg-gradient-to-r from-teal-50 to-green-50 p-5 rounded-xl mb-4 border border-teal-200">
                    <p class="text-lg font-bold text-teal-800">🔍 Распознанные объекты</p>
                    <div class="flex flex-wrap gap-2 mt-2">
                        ${uniqueObjects.length > 0 ? uniqueObjects.map(obj => `<span class="bg-teal-100 text-teal-800 px-3 py-1 rounded-full text-sm">${obj}</span>`).join('') : '<span class="text-gray-500">Ничего не найдено</span>'}
                    </div>
                    ${stainDetected ? '<p class="text-sm text-orange-600 mt-2">⚠️ Обнаружены возможные пятна / загрязнения</p>' : ''}
                </div>
                <h4 class="font-bold mb-2">✅ Рекомендуемые услуги:</h4>
                <div class="space-y-2">
                    ${recommended.map(s => `
                        <div class="flex justify-between items-center bg-white p-3 rounded-xl border hover:shadow-md">
                            <div><span class="font-semibold">${s.icon} ${s.name}</span><span class="text-teal-700 font-bold ml-2">от ${s.price.toLocaleString()} ₽${s.unit ? ' / ' + s.unit : ''}</span></div>
                            <button onclick="addToCart('${s.id}')" class="bg-teal-500 text-white px-3 py-1 rounded-lg text-sm hover:bg-teal-600">В корзину</button>
                        </div>
                    `).join('')}
                </div>
                <div class="mt-6 text-center border-t border-gray-200 pt-4">
                    <p class="text-2xl font-black text-gray-800 uppercase tracking-wider">Окончательная сумма после осмотра</p>
                    <p class="text-sm text-gray-500 mt-1">Точная стоимость будет согласована после осмотра объекта специалистом</p>
                </div>
            `;
            resultDiv.classList.remove('hidden');
        }

        async function submitOrder() {
            const name = document.getElementById('orderName').value.trim();
            const phone = document.getElementById('orderPhone').value.trim();
            if (!name || !phone) { showToast('❌ Заполните имя и телефон'); return; }
            if (!selectedDate) { showToast('❌ Выберите дату'); return; }

            const address = document.getElementById('orderAddress').value;
            const comment = document.getElementById('orderComment').value;
            const total = getCartTotal();
            const items = cart.map(item => {
                const s = getServiceById(item.id);
                return `${s.name} (${s.price} ₽${s.unit ? '/'+s.unit : ''}) × ${item.quantity}`;
            }).join('\n');

            const formData = new FormData();
            formData.append('name', name);
            formData.append('phone', phone);
            formData.append('address', address);
            formData.append('desiredDate', selectedDate);
            formData.append('desiredTime', selectedTime || 'Не указано');
            formData.append('comment', comment);
            formData.append('items', items);
            formData.append('total', total.toLocaleString() + ' ₽');
            formData.append('timestamp', new Date().toLocaleString());

            const photoInput = document.getElementById('orderPhotos');
            if (photoInput.files) { for (let file of photoInput.files) formData.append('photos', file); }

            const order = {
                id: Date.now(), date: new Date().toLocaleString(),
                name, phone, address,
                desiredDate: selectedDate, desiredTime: selectedTime || '',
                comment, items: [...cart], total, status: 'Отправлена'
            };
            orders.unshift(order);
            localStorage.setItem('orders', JSON.stringify(orders));
            if (currentUser) {
                currentUser.orders = orders.filter(o => o.phone === currentUser.phone);
                localStorage.setItem('currentUser', JSON.stringify(currentUser));
            }

            try {
                const response = await fetch(`https://formspree.io/f/${FORMSPREE_FORM_ID}`, {
                    method: 'POST', body: formData
                });
                if (response.ok) {
                    showToast('✅ Заявка отправлена на agentchistoty51@mail.ru!');
                } else {
                    const errorText = await response.text();
                    console.error('Ошибка Formspree:', response.status, errorText);
                    showToast(`⚠️ Ошибка отправки (${response.status}). Проверьте настройки формы.`);
                    alert(`Сервер вернул ошибку ${response.status}:\n${errorText}\n\nВозможные причины:\n- Форма не активирована (подтвердите email)\n- Неверный ID формы\n- Превышен лимит (50 писем/мес)\n- Письмо в спаме`);
                }
            } catch (err) {
                console.error('Сетевая ошибка:', err);
                showToast('⚠️ Нет связи с сервером. Проверьте интернет.');
                alert('Не удалось отправить заявку: ' + err.message);
            }

            cart = []; saveCart(); renderCart();
            selectedDate = null; selectedTime = null;
            ['orderName','orderPhone','orderAddress','orderComment'].forEach(id => document.getElementById(id).value = '');
            document.getElementById('orderPhotos').value = '';
            renderCalendar();
        }

        // ==================== ПРОФИЛЬ ====================
        function renderProfile() {
            const notLogged = document.getElementById('profileNotLogged');
            const logged = document.getElementById('profileLogged');
            if (currentUser) {
                notLogged.classList.add('hidden'); logged.classList.remove('hidden');
                document.getElementById('profileName').textContent = currentUser.name;
                document.getElementById('profilePhone').textContent = currentUser.phone;
                const historyDiv = document.getElementById('orderHistory');
                const userOrders = orders.filter(o => o.phone === currentUser.phone);
                if (userOrders.length === 0) historyDiv.innerHTML = '<p class="text-gray-400">Нет заказов</p>';
                else historyDiv.innerHTML = userOrders.map(o => `
                    <div class="bg-white p-3 rounded-xl border mb-2">
                        <p class="text-sm font-bold">${o.date} — ${o.total.toLocaleString()} ₽</p>
                        <p class="text-xs text-gray-500">${o.items.map(i=>getServiceById(i.id)?.name).join(', ')}</p>
                        <p class="text-xs">📅 ${o.desiredDate || '—'} в ${o.desiredTime || '—'}</p>
                        <p class="text-xs">Статус: ${o.status}</p>
                    </div>
                `).join('');
            } else {
                notLogged.classList.remove('hidden'); logged.classList.add('hidden');
            }
        }
        function login() {
            const name = document.getElementById('loginName').value.trim();
            const phone = document.getElementById('loginPhone').value.trim();
            if (!name || !phone) { showToast('Заполните поля'); return; }
            currentUser = { name, phone, orders: [] };
            localStorage.setItem('currentUser', JSON.stringify(currentUser));
            renderProfile();
            showToast('👋 Добро пожаловать, ' + name + '!');
        }
        function logout() {
            currentUser = null;
            localStorage.removeItem('currentUser');
            renderProfile();
            showToast('Вы вышли из кабинета');
        }

        function showToast(msg) {
            const toast = document.getElementById('toast');
            toast.textContent = msg; toast.classList.remove('hidden');
            setTimeout(() => toast.classList.add('hidden'), 3000);
        }

        // ==================== ИНИЦИАЛИЗАЦИЯ ====================
        window.onload = function() {
            renderServices();
            updateCartBadge();
            if (currentUser) renderProfile();
        };
